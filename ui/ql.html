                System.out.println("Nhap mang can sap xep: ");
                for(int i=0;i<n;i++){
                    a[i]=input.nextInt();

                    for(int j=i;j<n;j++){
                        if ( a[i] < a[j] ){

                            a[i]=a[j];
                            a[j]=tmp;
                        }
                    }

                    System.out.print(a[i]+" ");
